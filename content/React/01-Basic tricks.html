Loops in JSX with .map()


class Message extends Components {
    render() {
        const msgs = [
        {id:1, text: 'Greetings!'},
        {id:2, text: 'Goodbye!'},
        ]
        return (
            < ul>
        {msgs.map(m => < li >{m.text}< li /> )}
            < /ul> ) } }

Helpers.js : make functions available everywhere


function choice(arr){
    let randomIndex = Math.floor(Math.random() * arr.length)
    return arr[randomIndex] }
export { choice };

// then use :
