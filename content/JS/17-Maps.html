// set 
const question = new Map();
question.set('question', 'what is the official name of Javascript?');
question.set(1, 'es5')
question.set(2, 'es6')
question.set(3, 'es2015')
question.set(4, 'es7')
question.set('correct', 4)
question.set(true, 'correct answer')
question.set(false, 'wrong dude')

console.log(question)
/*[[Entries]]
0: {"question" => "what is the official name of Javascript?"}
1: {1 => "es5"}
2: {2 => "es6"}
3: {3 => "es2015"}
4: {4 => "es7"}
5: {"correct" => 4}
6: {true => "correct answer"}
7: {false => "wrong dude"} */

//get
console.log(question.get('question')) // what is the official name of Javascript?
console.log(question.get(4))// es7

//size

    //delete
