Arrays & Objects

Arrays


//Arrays are indexed:
let colors = ['red', 'orange', 'blue'];
colors.length /* 3 */ colors[0] /*'red' */ colors[1]/'orange'
  
// Modifying Arrays:
colors[0] = 'green';
colors[3] = 'purple;'
console.log(colors) // ['green', 'orange', 'blue','purple']

/*You'll often see people use const with arrays.
Values can change as long as the reference remains the same */
const myEggs = ['brown', 'brown'];
myEggs.push('purple');
myEggs[0] = 'green';
myEggs; //["green", "brown", "purple"]

Arrays Methods


/* Built-in methods return values when we call them.
We can then store them. */

// push(): add to end
let arr = [1,2,3];
arr.push(4)
console.log(arr) // [1,2,3,4]

// pop(): remove from end
let arr = [1,2,3,4];
arr.pop() // 4
console.log(arr) // [1,2,3]

// shift(): remove from start
let arr = [1,2,3];
arr.shift() //1
console.log(arr) [2,3]

// unshift(): add to start 
let arr = [2,3];
arr.unshift(1) //3
console.log(arr) // [1,2,3]

// concat(): merge arrays
let arr = [1,2,3];
let arr2 = [4,5,6]
let merged = arr.concat(arr2)
console.log(merged) // [1,2,3,4,5,6]

// includes(): Check if an array contains the specified element
let bibi = ['Alek', 'CuTo', 'Vung Tau'];
bibi.includes('Alek') // true

//indexOf(): Search the array for an element and returns its position
let bibi = ['Alek', 'CuTo', 'Vung Tau'];
bibi.indexOf('CuTo') // 1

// join(): Joins all elements of an array into a string
let num = [1,2,3,4,5];
num.join() // "1,2,3,4,5"

// reverse(): Reverses the order of the elements in an array
let num = [1,2,3,4,5];
num.reverse = [5,4,3,2,1]

// sort(): sorts the values as strings in alphabetical and ascending order.
var fruits = ["Banana", "Orange", "Apple", "Mango"];
fruits.sort(); //["Apple", "Banana", "Mango", "Orange"]

/* slice(): returns the selected elements in an array, as a new array object.
   selects the elements starting at the given start argument, and ends at, but does not include, 
   the given end argument.Note: The original array will not be changed.*/
var fruits = ["Banana", "Orange", "Lemon", "Apple", "Mango"];
var citrus = fruits.slice(1, 3); 
fruits // ["Banana", "Orange", "Lemon", "Apple", "Mango"]
citrus // ['Orange', 'Lemon']

/* splice(): adds/removes items to/from an array, and returns the removed item(s).
   array.splice(index, howmany, item1, ....., itemX) */
var fruits = ["Banana", "Orange", "Apple", "Mango"];
fruits.splice(2, 0, "Lemon", "Kiwi");
fruits // ["Banana", "Orange", "Lemon", "Kiwi", "Apple", "Mango"]

Array Callback methods

forEach()


/* Accepts a callback function.
   Calls the function once per element in the array.*/

const nums = [9,8,7,6,5,4,3,2,1];
nums.forEach(function (n) {console.log(n * n) })
// 64 49 36 25 16 9 4 1
nums.forEach(function (el) {
  if (el % 2 === 0) {console.log(el)} })
// 8 6 4 2

map()


/* Creates a new array with the results of calling a
   callback on every element in the array */

const texts = ['rofl', 'lol', 'omg']
const caps = texts.map(function (t) {return t.toUpperCase();})
caps // ["ROFL", "LOL", "OMG"]

filter()


/* Creates a new array with all elements that pass the test 
   implemented by the provided function. */
const nums = [9, 8, 7, 6, 5, 4, 3, 2, 1];
const odds = nums.filter(n => {return n % 2 === 1})
odds; //[9, 7, 5, 3, 1]
const smallNums = nums.filter(n => {return n >6})
smallNums // [9,8,7]

every()


/* tests whether all elements in the array pass the
   provided function. It returns a Boolean value. */
const words = ['dog', 'dig', 'log', 'bag'];
words.every(word => {return word.length === 3}); // true
words.every(word =>  word[0] === 'd') // false
words.every(w => {
  let lastLetter = w[w.length -1];
  return lastLetter === 'g' }) // true

some()


/* Similar to every, but returns true if ANY of
   the array elements pass the test function */
const words = ['dog', 'dig', 'log', 'bag', 'cupcake'];
words.some(word => {return word.length > 4}) //true
words.some(word => word[0] === 'z') //false
words.some(w => w.includes('cake')) // true

reduce()


/* Executes a reducer function on each element of the 
   array, resulting in a single value.*/
   
//arr.reduce(callback( accumulator, currentValue[, index[, array]] )[, initialValue])

//Summing an Array:
[3, 5, 7, 9, 11].reduce((accumulator, currentValue) => {
  return accumulator + currentValue;}) // 35

// Finding Max Val: 
let grades = [89, 96, 58, 77, 65];
const topScore = grades.reduce((max, currVal) => {
  if (currVal > max) return currVal;
  return max; }) 
topScore // 96
// a shorter option with Math.max & implicit return
const topScore = grades.reduce((max, currVal) =>
  Math.max(max, currVal))

// Initial value
[4, 5, 6, 7, 8].reduce((accumulator, currentValue) =>{
  return accumulator + currentValue; }, 100); // 130

// Tallying (calculate number of)
const fruitBasket = ['banana', 'cherry', 'orange', 'apple',
'cherry', 'orange', 'apple', 'banana', 'cherry', 'orange', 'fig' ];

const count = fruitBasket.reduce( (tally, fruit) => {
  tally[fruit] = (tally[fruit] || 0) + 1 ;
  return tally;
} , {})
count; // {banana: 2, cherry: 3, orange: 3, apple: 2, fig: 1}

// Flatenning an Array
const data = [[1, 2, 3], [4, 5, 6], [7, 8, 9]];

const flat = data.reduce((total, amount) => {
  return total.concat(amount);
}, []);

flat // [ 1, 2, 3, 4, 5, 6, 7, 8, 9 ]

sort() part II


/* If compareFunc(a,b) returns less than 0 Sort a before b
If compareFunc(a,b) returns 0 Leave a and b unchanged with 
respect to each other
If compareFunc(a,b) returns greater than 0 Sort b before a*/

const prices = [400.5, 3000, 99.99, 36, 12]
prices.sort(); // [12, 3000, 36, 400.5, 99.99];
prices.sort((a,b) => a -b ) // [12, 36, 99.99, 400, 3000];
prices.sort((a,b) => b - a) // [3000, 400.5, 99.99, 36, 12]

Method Description

entries() Returns a key/value pair Array Iteration Object

fill() Fill the elements in an array with a static value
filter() Creates a new array with every element in an array that pass a test
find() Returns the value of the first element in an array that pass a test
findIndex() Returns the index of the first element in an array that pass a test
forEach() Calls a function for each array element
from() Creates an array from an object
includes() Check if an array contains the specified element
indexOf() Search the array for an element and returns its position
isArray() Checks whether an object is an array
join() Joins all elements of an array into a string
keys() Returns a Array Iteration Object, containing the keys of the original array
lastIndexOf() Search the array for an element, starting at the end, and returns its position
map() Creates a new array with the result of calling a function for each array element
pop() Removes the last element of an array, and returns that element
push() Adds new elements to the end of an array, and returns the new length
reduce() Reduce the values of an array to a single value (going left-to-right)
reduceRight() Reduce the values of an array to a single value (going right-to-left)
reverse() Reverses the order of the elements in an array
shift() Removes the first element of an array, and returns that element
slice() Selects a part of an array, and returns the new array
some() Checks if any of the elements in an array pass a test
sort() Sorts the elements of an array
splice() Adds/Removes elements from an array
toString() Converts an array to a string, and returns the result
unshift() Adds new elements to the beginning of an array, and returns the new length
