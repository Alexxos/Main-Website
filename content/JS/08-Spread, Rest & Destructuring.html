Spread operator


/* Spread  for Function calls  
Expands an iterable (array, string, etc.) into list of arguments: */
function addFourAges (a,b,c,d) { return a + b + c + d  }
const ages = [18, 20, 30, 24];
let sum = addFourAges(...ages); // ... spread the array into variable
console.log(sum) // 92

/* Spread in Array Litterals:   
Create a new array using an existing array. Spreads the elements from one array into a new array. */

let num1 = [1, 2, 3,];
let num2 = [4, 5, 6];
let num3;
[...num1, ...num2];// [1, 2, 3, 4, 5, 6]
[...num1, 'a', 'b', ...num2]; //[1, 2, 3, 'a', 'b', 4, 5, 6];
num3 = [...num1] // num3.push(4) = [1, 2, 3, 4]

/* Spread in Object Litterals 
Copies properties from on object into another Object literal */

const feline = { legs:4, Family:'Felidae' };
const canine = { family: 'Caninae', furry:true };

const dog = { ...canine, isPet:true };
//{ family: 'Caninae', furry: true, isPet:true}
const lion = { ...feline, genus: 'Panthera' };
//{ legs:4, family: 'Felidae', genus: 'Panthera' }
const catDog = { ... feline, ... canine};
// { legs:4, family: 'Caninae',, furry: true }

Destructuring


/*A short, clean syntax to 'unpack': Values from arrays, Properties from objects, 
into distinct variable. */

// Destructuring Arrays : 
const raceResults = ['Eli Kipchoge', 'Feyisa Lelisa', 'Galen Rupp'];
const [gold, silver,bronze] = raceResults;
gold; /* 'Eli Kipchoge' */ silver; /* 'Feyisa Lelisa' */ bronze; // 'Galen Rupp'
const [ fastest, ...everyoneElse] = raceResults; 
fastest; /*'Eli Kipchoge'*/ everyoneElse // ['Feyisa Lelisa', 'Galen Rupp']

            
let [nickName1, nickName2] = ['Momo', 'weshwesh'];
console.log(nickName1, nickName2) // MoMo weshwesh

// Destructuring Objects :  
const personalInformation = {
firstName: 'Alek',
lastName: 'CuTo',
city: 'Vung Tau',
country: 'Vietnam'      };
    
const {firstName: fn, lastName: ln} = personalInformation;
console.log(`${fn} ${ln}`); // Alek CuTo

// Destructuring Parameters: 
const fullName = ({first, last}) => {return `${first} ${last}`}
const runner = { first: 'Eli', last: 'Kipchoge', country: 'Kenya' };
fullName(runner); // Eli Kipchoge
