Objects, Methods and This


/* Objects are collections of properties. Properties are a key-value pair. Rather than accessing
data using an index, we use custom keys.*/

const fitBitData = {
  totalSteps: 302030,
  totalMiles: 220.6,
  workoutsThisWeek: '5 out of 7',
};

//Returning properties 
fitBitData.totalSteps // 302030

// Updating properties:
fitBitData.workoutsThisWeek = '6 of 7';
fitBitData.totalMiles += 7.5;

// adding a new property
fitBitData.heartStillBeating = true;



//Values can change
